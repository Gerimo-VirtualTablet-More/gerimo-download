#!/usr/bin/env bash
set -euo pipefail

# HIER deinen bestehenden Repo-GPG-Key eintragen (Name oder Mail aus `gpg --list-secret-keys`)
GPG_ID="Gerimo <antozstudios@proton.me>"

# Script immer in das Verzeichnis wechseln, in dem es liegt
cd "$(dirname "$0")"

echo "[1/4] Packages neu erzeugen..."
dpkg-scanpackages pool/ /dev/null > Packages
gzip -9 < Packages > Packages.gz

echo "[2/4] Release neu erzeugen..."
apt-ftparchive release . > Release

echo "[3/4] Release signieren (InRelease)..."
gpg --yes --clearsign --local-user "$GPG_ID" -o InRelease Release

echo "[4/4] Release.gpg erzeugen..."
gpg --yes --detach-sign --local-user "$GPG_ID" -o Release.gpg Release

echo "Fertig. Jetzt noch git add/commit/push (von diesem Ordner aus):"
echo "  git add Packages Packages.gz Release InRelease Release.gpg"
echo "  git commit -m \"Update APT repo metadata\""
echo "  git push"
